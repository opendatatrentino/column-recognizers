# 	REGEX, REGEX_S, HEADER_REGEX, VALUE_SET, TF_IDF, SVM, HEADER, SUM_THRESHOLD, ONE_BEST

# SVM			model file path; component recognizer name, ...

# HEADER		none
# ONE_BEST		none
#
